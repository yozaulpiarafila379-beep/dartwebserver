1. Class Hewan (Polimorfisme)

Kode ini mendemonstrasikan polimorfisme dan abstraksi. Kelas abstrak Hewan memiliki metode bersuara() yang belum diimplementasikan. Subkelas seperti Kucing, Anjing, dan Burung kemudian menyediakan implementasi unik untuk bersuara(), menunjukkan perilaku yang berbeda dari metode yang sama.

Kode Dart:


// ==========================================
// Nama  : Yoza Ulpia Rafila
// NIM   : 251410004
// Kelas : SI2A
// ==========================================

abstract class Hewan {
  String nama = "Hewan";

  void bersuara();
}

class Kucing extends Hewan {
  Kucing() {
    nama = "Kucing";
  }

  @override
  void bersuara() {
    print('$nama: Meong');
  }
}

class Anjing extends Hewan {
  Anjing() {
    nama = "Anjing";
  }

  @override
  void bersuara() {
    print('$nama: Guk guk');
  }
}

class Burung extends Hewan {
  Burung() {
    nama = "Burung";
  }

  @override
  void bersuara() {
    print('$nama: Cuit cuit');
  }
}

void main() {
print("=== Tugas Polimorfisme & Abstraksi ===");

print("NIM   : 251410004");
print("Kelas : SI2A");

3. Class Kendaraan (Operator `is` & `as`)

Kode ini menunjukkan cara menggunakan operator is untuk memeriksa tipe objek pada saat runtime dan operator as untuk melakukan type casting atau perubahan tipe data. Penggunaan as? untuk safe casting mencegah error jika tipe tidak cocok.

Kode Dart:


// ==========================================
// Nama  : Yoza Ulpia Rafila
// NIM   : 251410004
// Kelas : SI2A
// ==========================================

class Kendaraan {}

class Mobil extends Kendaraan {
  void klakson() {
    print('Tin!');
  }
}

void main() {
print("=== Operator is dan as ===");
print("Nama  : Yoza Ulpia Rafila");
print("NIM   : 251410004");
print("Kelas : SI2A");
print("--------------------------");

  Kendaraan k = Mobil();

  // Mengecek tipe dengan "is"
  if (k is Mobil) {
    print("Cek tipe: k adalah Mobil");
    k.klakson();
  }

  // Safe casting dengan "as?"
  print("Safe casting:");
  (k as Mobil?)?.klakson();
}

4. Abstract Class Kendaraan

Contoh abstraksi. Kelas Kendaraan dideklarasikan sebagai abstrak dan memiliki metode abstrak start(), yang berarti kelas ini tidak bisa diinstansiasi langsung dan start() harus diimplementasikan oleh kelas turunan (contohnya Mobil). Metode stop() adalah konkret dan memiliki implementasi langsung.

Kode Dart:


// ==========================================
// Nama  : Yoza Ulpia Rafila
// NIM   : 251410004
// Kelas : SI2A
// ==========================================

abstract class Kendaraan {
  String merk;

  // Constructor
  Kendaraan(this.merk);

  // Method abstract
  void start();

  // Method concrete
  void stop() {
    print('Kendaraan berhenti');
  }
}

class Mobil extends Kendaraan {
  // Constructor turunan
  Mobil(String merk) : super(merk);

  @override
  void start() {
    print('Mobil $merk menyala');
  }
}

void main() {
print("=== Abstract Class Kendaraan ===");
print("Nama  : Yoza Ulpia Rafila");
print("NIM   : 251410004");
print("Kelas : SI2A");
print("--------------------------------");

  Mobil mobil = Mobil("Toyota");

  mobil.start();
  mobil.stop();
}

6. Class Pesawat (Interface)

Menunjukkan konsep interface dalam Dart. Kelas Pesawat mengimplementasikan Kendaraan dan Terbang (yang keduanya adalah kelas abstrak yang bertindak sebagai interface). Ini berarti Pesawat harus menyediakan implementasi untuk semua metode yang didefinisikan dalam kedua "interface" tersebut.

Kode Dart:


// ==========================================
// Nama  : Yoza Ulpia Rafila
// NIM   : 251410004
// Kelas : SI2A
// ==========================================

// Abstract Class

  void jalan();
}

// Interface
abstract class Terbang {
  void terbang();
}

// Class Pesawat
class Pesawat extends Kendaraan implements Terbang {
  @override
  void jalan() {
    print('Pesawat berjalan di landasan');
  }

  @override
  void terbang() {
    print('Pesawat terbang di udara');
  }
}

print("Nama  : Yoza Ulpia Rafila");
print("NIM   : 251410004");
print("Kelas : SI2A");

14. Utility Math (Bonus 5)

Kumpulan fungsi matematika statis seperti faktorial, prima, rata-rata, dan median.

Kode Dart:


// ==========================================
// Nama  : Yoza Ulpia Rafila
// NIM   : 251410004
// Kelas : SI2A
// ==========================================
/// Bonus 5

import 'dart:math';

class MathUtils {

  static Random rand = Random(1);

  static int faktorial(int n) {
    if(n <= 1) return 1;
    return n * faktorial(n-1);
  }

  static bool cekPrima(int n) {
    if(n <= 1) return false;
    for(int i = 2; i <= sqrt(n); i++) {
      if(n % i == 0) return false;
    }
    return true;
  }

  static double pembulatan(double angka, int desimal) {
    double scale = pow(10, desimal).toDouble();
    return (angka * scale).round() / scale;
  }

  static int randomInt(int min, int max) {
    return min + rand.nextInt(max - min + 1);
  }

  static double rataRata(List list) {
    int total = list.reduce((a, b) => a + b);
    return total / list.length;
  }

  static double median(List list) {
    List sorted = List.from(list)..sort();
    int n = sorted.length;

    if(n % 2 == 0) {
      return (sorted[n ~/ 2 - 1] + sorted[n ~/ 2]) / 2;
    } else {
      return sorted[n ~/ 2].toDouble();
    }
  }
}

void main() {
  print("=== Utility Math ===");
  print("Nama  : Yoza Ulpia Rafila");
  print("NIM   : 251410004");
  print("Kelas : SI2A");
  print("--------------------------------");

  print("Faktorial 5 = ${MathUtils.faktorial(5)}");
  print("Prima 7 = ${MathUtils.cekPrima(7)}");
  print("Pembulatan = ${MathUtils.pembulatan(3.14159, 2)}");
  print("Contoh angka = ${MathUtils.randomInt(1, 10)}");
  print("Rata-rata = ${MathUtils.rataRata([10, 20, 30, 40, 50])}");
  print("Median = ${MathUtils.median([10, 20, 30, 40, 50])}");
}
/*
Hasil Output:
Faktorial 5 = 120
Prima 7 = true
Pembulatan = 3.14
contoh angka = 3
Rata-rata = 30
Median = 30
*/

15. Kesimpulan

Dari latihan-latihan di atas, dapat disimpulkan bahwa:

  • Abstraksi menyembunyikan detail implementasi dan hanya menampilkan fungsi penting kepada pengguna.
  • Polimorfisme memungkinkan kita menggunakan satu nama method untuk melakukan berbagai tindakan yang berbeda tergantung objeknya.
  • Inheritance (Pewarisan) memungkinkan pembuatan class baru berdasarkan class yang sudah ada, sehingga kode lebih efisien dan terstruktur.
  • Interface berfungsi sebagai kontrak yang memaksa class turunan untuk mengimplementasikan semua method yang didefinisikan di dalamnya.
  • Dengan menggabungkan konsep-konsep OOP (Object-Oriented Programming) ini, program menjadi lebih mudah dikembangkan, dipelihara, dan dimodifikasi di kemudian hari.
